🎬 BoTTube Embed Player

Usage


const player = new BoTTubePlayer('player', {
  videoId: 'VIDEO_ID',
  width: 640,
  height: 360,
  autoplay: false
});

// Controls
player.play();
player.pause();
player.seek(30);
player.setVolume(80);

HTML Embed


<div id="player" 
     data-bottube-player="VIDEO_ID"
     data-width="640"
     data-height="360"
     data-autoplay="false">
</div>
<script src="embed-player.js"></script>
